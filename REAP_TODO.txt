-- From Session 5D: New tables for deal improvements and units
CREATE TABLE deal_improvements (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  deal_id uuid NOT NULL REFERENCES deals(id) ON DELETE CASCADE,
  description text NOT NULL,
  budgeted_cost numeric,
  actual_cost numeric,
  status text DEFAULT 'Not Started',
  contractor text,
  created_at timestamptz DEFAULT now()
);

CREATE TABLE deal_units (
  id uuid DEFAULT gen_random_uuid() PRIMARY KEY,
  deal_id uuid NOT NULL REFERENCES deals(id) ON DELETE CASCADE,
  unit_number text NOT NULL,
  sqft numeric,
  bedrooms integer,
  bathrooms numeric,
  current_rent numeric,
  market_rent numeric,
  status text DEFAULT 'Vacant',
  tenant_name text,
  created_at timestamptz DEFAULT now()
);

ALTER TABLE deal_improvements ENABLE ROW LEVEL SECURITY;
ALTER TABLE deal_units ENABLE ROW LEVEL SECURITY;
CREATE POLICY "Authenticated users can manage improvements" ON deal_improvements FOR ALL USING (auth.role() = 'authenticated');
CREATE POLICY "Authenticated users can manage units" ON deal_units FOR ALL USING (auth.role() = 'authenticated');
